import os
from os import getcwd, chdir
from os.path import isdir, join
from subprocess import call
from time import time



[docs]class GitRepo:
    def __init__(self, directory, url, branch, update_freq=1, tag=None):
        self.dir = directory
        self.url = url
        self.branch = branch
        self.update_freq = update_freq
        self.tag = tag


[docs]    def run_inside(self, command):
        cur_path = getcwd()
        try:
            chdir(self.dir)
            if isinstance(command, str):
                call(command.split(' '))
            else:
                call(command)
        finally:
            chdir(cur_path)


    def _clone(self):
        if self.tag is None:
            call(['git', 'clone', '-b', self.branch, '--single-branch', '--depth', '1', self.url, self.dir])
        else:
            call(['git', 'init'])
            call(['git', 'remote', 'add', 'origin', self.url])
            call(['git', 'fetch', 'origin', self.tag, '--depth', '1'])
            call(['git', 'reset', '--hard', 'FETCH_HEAD'])


[docs]    def try_pull(self):
        if not isdir(self.dir):
            self._clone()
            return True

        git_folder = join(self.dir, '.git')
        stat = os.stat(git_folder)
        if time() - stat.st_mtime > self.update_freq * 60 * 60:
            # Touch folder
            os.utime(git_folder)
            if not self.tag:
                self.run_inside(['git', 'pull', '--ff-only'])
            return True
        return False
